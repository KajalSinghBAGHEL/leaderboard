22   /**
23    * Displays all time data of each knolder on API.
24    *
25    * @param scoreForBlogsPerAuthor List of GetAuthorScore case class objects, which specifies full name,
26    *                               overall score and overall rank of each knolder.
27    * @return Http request binded with server port.
28    */
29   def pushAllTimeDataOnAPI(scoreForBlogsPerAuthor: List[GetAuthorScore]): Future[Http.ServerBinding] = {
30     val reputation = compactRender(decompose(scoreForBlogsPerAuthor))
31     val route =
32       path("reputation") {
33         get {
34           complete(HttpEntity(ContentTypes.`application/json`, reputation))
35         }
36       }
37     Http().bindAndHandle(route, config.getString("interface"), config.getInt("port"))
38   }.recoverWith {
39     case ex: Exception => Future.failed(new Exception("Service failed", ex))
40   }
