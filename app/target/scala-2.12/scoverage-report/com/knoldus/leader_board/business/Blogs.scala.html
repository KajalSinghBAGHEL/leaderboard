78   /**
79    * Hits wordpress API page wise and unmarshalls the response.
80    *
81    * @param lastPage Specifies end page number upto which wordpress API needs to be hit from first page.
82    * @return BlogAuthor case class object which contains list of all blogs and knolders.
83    */
84   private def getAllBlogs(lastPage: Int): Future[BlogAuthor] = {
85     @scala.annotation.tailrec
86     def getBlogs(blogsAndAuthorsList: Future[BlogAuthor], currentPage: Int, end: Int): Future[BlogAuthor] = {
87       if (currentPage > end) {
88         blogsAndAuthorsList
89       } else {
90         val response: Future[HttpResponse] = dispatchRequest(
91           HttpRequest(uri = config.getString("uri") + "?page=" + currentPage))
92         getBlogs(blogsAndAuthorsList.flatMap(blogsAndAuthorsList =>
93           response.flatMap(res => Unmarshal(res.entity).to[String])
94             .map(blogs =>
95               BlogAuthor(blogsAndAuthorsList.blogs ::: getListOfBlogs(blogs),
96                 (blogsAndAuthorsList.authors ::: getListOfAuthors(blogs)).distinct))), currentPage + 1, end)
97       }
98     }
99 
100     val blogsAndAuthorsList = getBlogs(Future {
101       BlogAuthor(List.empty, List.empty)
102     }, 1, lastPage)
103     blogsAndAuthorsList
104   }.recoverWith {
105     case ex: Exception => Future.failed(new Exception("Service failed", ex))
106   }

114   /**
115    * Extracts blogs data from wordpress API.
116    * Extracts only those blogs which are not yet added in blog table.
117    *
118    * @param blogs JSON string of blogs data fetched from wordpress API.
119    * @return List of Blog case class objects, which specifies blog id, wordpress id of knolder, date of blog,
120    *         title of blog.
121    */
122   private def getListOfBlogs(blogs: String): List[Blog] = {
123     val parsedBlogs = parse(blogs)
124     val listOfBlogs = (parsedBlogs \ "posts").children map { blog =>
125       val blogId = (blog \ "ID").extract[Option[Int]]
126       val authorLogin = (blog \ "author" \ "login").extract[Option[String]]
127       val date = (blog \ "date").extract[String]
128       val odtInstanceAtOffset = ZonedDateTime.parse(date)
129       val odtInstanceAtUTC = odtInstanceAtOffset.withZoneSameInstant(ZoneOffset.UTC)
130       val publishedOn = Timestamp.from(odtInstanceAtUTC.toInstant)
131       val title = (blog \ "title").extract[Option[String]]
132       val fetchMaxDate = fetchData.fetchMaxBlogPublicationDate
133       val minimumTime = Instant.parse("0000-04-01T09:37:10Z")
134       val parsedDate = Timestamp.from(minimumTime)
135       fetchMaxDate match {
136         case Some(value) => val date = value
137           if (publishedOn.compareTo(date) == 1) {
138             Blog(blogId, authorLogin, publishedOn, title)
139           }
140           else {
141             Blog(None, None, parsedDate, None)
142           }
143         case None =>
144           if (publishedOn.compareTo(parsedDate) == 1) {
145             Blog(blogId, authorLogin, publishedOn, title)
146           }
147           else {
148             Blog(None, None, parsedDate, None)
149           }
150       }
151     }
152     listOfBlogs.filter(blog => blog.blogId.isDefined && blog.authorLogin.isDefined)
153   }
