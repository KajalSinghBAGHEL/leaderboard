19       listOfScores match {
20         case Nil => rankList
21         case _ :: Nil => rankList

32               rankListIndex + 1)
33           }
34           else {
35             getRank(second :: rest, rankList :+ GetRank(second.knolderId, rankList(rankListIndex).rank),
36               rankListIndex + 1)
37           }
38       }
39     }
40 
41     val firstKnolderId = listOfScores match {
42       case Nil => 0
43       case knolder :: Nil => knolder.knolderId
44       case knolder :: _ => knolder.knolderId
45     }
46     getRank(listOfScores, rankList :+ GetRank(firstKnolderId, 1), 0)
47   }
