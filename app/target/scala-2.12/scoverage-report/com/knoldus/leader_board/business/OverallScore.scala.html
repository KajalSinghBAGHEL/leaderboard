8   /**

10    *
11    * @param listOfBlogsAndAuthors BlogAuthor case class object which contains list of all blogs and list of
12    *                              all knolders.
13    * @return List of AuthorScore case class objects specifying wordpress id, name and overall score of each knolder.
14    */
15   def calculateScore(listOfBlogsAndAuthors: BlogAuthor): List[AuthorScore] = {
16     val numberOfBlogsPerKnolder: List[BlogCount] = listOfBlogsAndAuthors.authors.map { author =>
17       BlogCount(author.authorLogin, author.authorName, listOfBlogsAndAuthors.blogs
18         .count(blog => blog.authorLogin == author.authorLogin))
19     }
20     numberOfBlogsPerKnolder.map {
21       blogs => AuthorScore(blogs.authorLogin, blogs.authorName, blogs.numberOfBlogs * 5, 0)
22     }
23   }
24 
25   /**
26    * Verifies whether knolder already exist in all_time table or not, if not then it inserts it into the table with its
27    * overall score, if it does then it updates its overall score.
28    *
29    * @param listOfScores List of AuthorScore case class objects which contains score of each knolder.
30    * @return List of Integers which display the status of query execution.
31    */
32   def manageScores(listOfScores: List[AuthorScore]): List[Int] = {
33     listOfScores.map { scores =>
34       val authorId = fetchData.fetchKnolderIdFromKnolder(scores)
35       val knolderId = fetchData.fetchKnolderIdFromAllTime(scores, authorId)
36       knolderId match {
37         case Some(_) => updateData.updateAllTimeData(scores, authorId)
38         case None => storeData.insertAllTimeData(scores, authorId)
39       }
40     }
41   }
42 }
